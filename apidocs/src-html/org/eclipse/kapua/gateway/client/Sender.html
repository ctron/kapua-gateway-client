012package org.eclipse.kapua.gateway.client;
013
014import static java.util.Objects.requireNonNull;
015
016import java.util.Optional;
017
018public interface Sender<X extends Throwable> {
019
020    public void send(Payload payload) throws X;
021
022    public default void send(final Payload.Builder payload) throws X {
023        send(payload.build());
024    }
025
026    public default <Y extends Throwable> Sender<Y> errors(final ErrorHandler<Y> errorHandler) {
027        return new Sender<Y>() {
028
029            @Override
030            public void send(final Payload payload) throws Y {
031                requireNonNull(payload);
032
033                try {
034                    Sender.this.send(payload);
035                } catch (final Throwable e) {
036                    errorHandler.handleError(e, Optional.of(payload));
037                }
038            }
039        };
040
041    }
042}
